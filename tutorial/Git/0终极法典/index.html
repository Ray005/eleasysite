    git init #初始化
#（修改文件后进行下一步）
git add -A  #添加修改的文件
git status  #查看一下更改状态
git commit -am '提交信息'   #提交
git remote add origin git@github.com:ray005/example.git #关联github远程库并命名为origin
git push -u origin main #推送到远程库origin下的main分支
